<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Interactive Wood Map</title>
    <link rel="stylesheet" href="https://unpkg.com/leaflet@1.7.1/dist/leaflet.css" />
    <script src="https://unpkg.com/leaflet@1.7.1/dist/leaflet.js"></script>
    <style>
        #map { height: 500px; width: 100%; }
    </style>
</head>
<body>
    <div id="map"></div>
    <script>
        var map = L.map('map').setView([20, 0], 2);

        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
            attribution: '&copy; OpenStreetMap contributors'
        }).addTo(map);

        var woods = [
            { name: "Larch", location: [50, 10], description: "Larch is a durable softwood from Europe & USA, used for playground equipment.", image: "https://example.com/larch.jpg" },
            { name: "Douglas Fir", location: [48, -120], description: "Douglas Fir is widely used in Europe and North America for structural elements.", image: "https://example.com/douglas_fir.jpg" },
            { name: "Robinia", location: [47, 15], description: "Robinia is a highly durable wood from Europe, often used in high-stress applications.", image: "https://example.com/robinia.jpg" },
            { name: "Bilinga", location: [0, 15], description: "Bilinga is a tropical hardwood from West & Central Africa, highly durable and wear-resistant.", image: "https://example.com/bilinga.jpg" }
        ];

        woods.forEach(function(wood) {
            var marker = L.marker(wood.location).addTo(map);
            marker.bindPopup(`<b>${wood.name}</b><br>${wood.description}<br><img src='${wood.image}' width='150'>`);
        });
    </script>
</body>
